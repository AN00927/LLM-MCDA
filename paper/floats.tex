\section*{Item 2 — MAVT Equations Block}

\noindent These three equations appear inline in the Methodology body (not as floats).

\begin{equation}
  \label{eq:mavt_sum}
  s_j = \sum_{i=1}^{4} w_i \cdot v_i(x_{ij})
\end{equation}

\begin{equation}
  \label{eq:linear_vf}
  v_i(x) = \frac{x - x_{\min}}{x_{\max} - x_{\min}} \times 10
\end{equation}

\begin{equation}
  \label{eq:log_vf}
  v_i(x) = \frac{\ln\!\left(1 + \alpha \cdot
  \dfrac{x - x_{\min}}{x_{\max} - x_{\min}}\right)}
  {\ln(1 + \alpha)} \times 10
\end{equation}
